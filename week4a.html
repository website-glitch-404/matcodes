function fut = program1(ut)

rhop = 1800;
rho = 994.6;
g = 9.80665;
Dp = 0.208*10^(-3);
myu = 8.931*10^(-4);

Re = (rho*Dp*ut)/myu;

if Re < 0.1
    cd = 24/Re;
else
    cd = (24/Re)*(1+0.14*Re^0.7);
end

fut = (ut^2*3*cd*rho) - (4*g*(rhop-rho)*Dp);
