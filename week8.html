tc2 = 328;
th1 = 388;

Ph = 950;

Cpc = 4187;
Cph = 2930;

U = 3490;

Vc = 15/3600;
Vh = 21/3600;

Mc = Vc*Pc;
Mh = Vh*Ph;

Qc = Mc*Cpc*(tc2-tc1);

th2 = th1 - (Qc/(Mh*Cph));

delt1 = th1-tc2;
delt2 = th2-tc1;

deltlmtdc = (delt2-delt1)/log(delt2/delt1);

Ac = Qc/(U*deltlmtdc);

delT1 = th1-tc1;
delT2 = th2-tc2;

delTlmtdp = (delT2-delT1)/log(delT2/delT1);

Ap = Qc/(U*delTlmtdp);
