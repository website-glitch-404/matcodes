WEEK 5 RK EOS

clc;
clear;
t = 473.15;
p = 12;
tc = 425.1;
pc = 37.96;
r = 83.14;
a = 0.42748*(r^2*tc^2.5)/(pc*t^0.5);
b = 0.08664*(r*tc/pc);
A = a*p/(r^2*t^2);
B = b*p/(r*t);
q = A-B-B^2;
R = -A*B;
f = q-1/3;
g = (27*R+9*q-2)/27;
C = (g^2)/4 + (f^3)/27;
if C>0
D=(-g/2+sqrt(C))^(1/3);
E=(-g/2-sqrt(C))^(1/3);
Z=D+E+1/3;
end
v=(r*t*Z)/p;
fprintf('Z=%0.4f\n',Z)
fprintf('v=%0.4f\n',v)

WEEK 9 Heat Conduction ODE45

function Tdot = abcd(t,T)
T1=0;T2=T(1);T3=T(2);T4=T(3);T5=T(4);T6=T(5);T7=T(6);T8=T(7);T9=T(8);T10=T(9);
A=2e-5;delx=0.1;
dT2dt=(A/delx^2)*(T3-2*T2+T1);
dT3dt=(A/delx^2)*(T4-2*T3+T2);
dT4dt=(A/delx^2)*(T5-2*T4+T3);
dT5dt=(A/delx^2)*(T6-2*T5+T4);
dT6dt=(A/delx^2)*(T7-2*T6+T5);
dT7dt=(A/delx^2)*(T8-2*T7+T6);
dT8dt=(A/delx^2)*(T9-2*T8+T7);
dT9dt=(A/delx^2)*(T10-2*T9+T8);
dT10dt=(A/delx^2)*(T9-T10);
Tdot=[dT2dt;dT3dt;dT4dt;dT5dt;dT6dt;dT7dt;dT8dt;dT9dt;dT10dt];

function Ydot = program1(x2,Y)
L=Y(1);
T=Y(2);
Ptotal=1.2*760;
A1=6.90565;B1=1211.033;C1=220.79;
A2=6.95464;B2=1344.8;C2=219.482;
P1=10^(A1-(B1/(T+C1)));
P2=10^(A2-(B2/(T+C2)));
K1=P1/Ptotal;
K2=P2/Ptotal;
x1=1-x2;
E=1-(K1*x1)*(K2*x2);
dldx2=L/(x2*(K2-1));
Kc=0.5*10^6;
dtdx2=Kc*E;
Ydot=[dldx2;dtdx2];
