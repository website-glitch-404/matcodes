function fz = program1(z)

T = 373.15;
P = 101325;
Tc = 647.4;
Pc = 221.9*10^5;
R = 8.314;

a = 0.42748*R^2*Tc^2.5/(Pc*T^0.5);
b = 0.08664*R*Tc/Pc;

A = (a*P)/(R^2*T^2);
B = (b*P)/(R*T);

fz = z^3 - z^2 + (A-B-B^2)*z - A*B;
