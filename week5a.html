clc;
clear all;

t = 473.15;
p = 12;
tc = 425.1;
pc = 37.96;
r = 83.14;

a = 0.42748*r^2*tc^2.5/(pc*t^0.5);
b = 0.08664*r*tc/pc;

A = a*p/(r^2*t^2);
B = b*p/(r*t);

R = -A*B;
q = A-B-B^2;

f = q-(1/3);
g = (27*R+9*q-2)/27;

C = (f^3/27)+(g^2/4);

D = ((-g/2)+sqrt(C))^(1/3);
E = ((-g/2)-sqrt(C))^(1/3);

k = (-f^3/27)^(1/2);
x = acos((-g/2)/k);

if C>0
    Z = D+E+(1/3);
else
    Z1 = 2*(k^(1/3))*cos(x/3)+(1/3);
    Z2 = 2*(k^(1/3))*cos((2*pi+x)/3)+(1/3);
    Z3 = 2*(k^(1/3))*cos((4*pi+x)/3)+(1/3);
    Z = max([Z1 Z2 Z3]);
end

v = (r*t*Z)/p;

fprintf('compressibility factor of n-butane at 473.15 k is %0.4f \n',Z)
fprintf('molar volume of n-butane is %0.4f \n',v)
