clc;
clear all;

uto = 4;

ut = fzero(@program1,uto);

rhop = 1800;
rho = 994.6;
g = 9.80665;
Dp = 0.208*10^(-3);
myu = 8.931*10^(-4);

Re = (rho*Dp*ut)/myu;

if Re < 0.1
    cd = 24/Re;
else
    cd = (24/Re)*(1+0.14*Re^0.7);
end

fprintf('the terminal velocity of the falling particle is %f m/s \n',ut)
fprintf('the reynolds number at terminal velocity of falling particle is: %f \n',Re)
fprintf('the drag coefficient of falling particle at terminal velocity is %f \n',cd)
