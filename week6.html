Compressibility Factor Variation from Van der Waals Equation of State

clc;
clear all;

Tc = 304.2;
Pc = 73.83;
Tr = 1.2;

T = Tc*Tr;

R = 83.14;

a = (27*R^2*Tc^2)/(64*Pc);
b = (R*Tc)/(8*Pc);

fprintf('Pr      Z \n')

for Pr = 0.1:0.1:10

    P = Pc*Pr;

    A = 1;
    B = -(b + (R*T/P));
    C = a/P;
    D = -(a*b)/P;

    S = roots([A B C D]);

    for j = 1:3
        if imag(S(j)) == 0
            V = S(j);
        end
    end

    Z = (P*V)/(R*T);

    fprintf('%1.4f   %1.4f \n',Pr,Z)

    plot(Pr,Z,'b.')
    hold on

end

xlabel('reduced pressure of CO2, Pr')
ylabel('compressibility factor of CO2, Z')
title('Variation of Z with Pr')
legend('Tr = 1.2')
