function Ydot = program1(x2,Y)

L = Y(1);
T = Y(2);

Ptotal = 1.2*760;

A1 = 6.90565;
B1 = 1211.033;
C1 = 220.79;

A2 = 6.95464;
B2 = 1344.8;
C2 = 219.482;

P1 = 10^(A1-(B1/(T+C1)));
P2 = 10^(A2-(B2/(T+C2)));

K1 = P1/Ptotal;
K2 = P2/Ptotal;

x1 = 1-x2;

E = 1-(K1*x1)*(K2*x2);

dldx2 = L/(x2*(K2-1));

Kc = 0.5*10^6;

dtdx2 = Kc*E;

Ydot = [dldx2;dtdx2];
