Calculation of the Flow Rate in a Pipe Line Using Successive Substitution

clc;
clear;

T = 60;
P1 = 150;
Z1 = 0;
P2 = 0;
Z2 = 300;
L = 1000;
D = 7.981/12;

G = 32.174;
Gc = 32.174;

ROW = 62.3539;
MUE = 0.76*10^(-3);

e = 0.00015;

delP = (P2-P1)*144;
delZ = Z2-Z1;

U(1) = 10;

for i = 1:100

    Re = (ROW*D*U(i))/MUE;

    if Re < 2100
        Ff = 16/Re;
    else
        Ff = 1/(16*(log10(((e/D)/3.7)-((5.02/Re)*log10(((e/D)/3.7)+(14.5/Re)))))^2);
    end

    U(i+1) = sqrt(((G*delZ)+(Gc*delP/ROW))/(0.5-(2*Ff*L/D)));

    tol = 10^(-5);

    if abs(U(i)-U(i+1)) < tol
        U = U(i+1);
        break;
    else
        U(i) = U(i+1);
    end

end

A = ((22/7)/4)*D^2;

q = U*A*7.4805*60;

fprintf('the volumetric flow rate is: %0.2f gallons per minute',q)
