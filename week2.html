n = 11.3*1000;
Q = 422*(10^6);
T1 = 533.15;

A = 3.578;
B = 3.020*10^(-3);
D = (-0.186)*10^(5);

R = 8.314;

delH = Q/n;

Tguess = 733.15;

Cpm = A + (B*(Tguess+T1)/2) + (D/(Tguess*T1));

T2guess = ((delH/R)/Cpm) + T1;

tol = 10^(-6);

while abs(T2guess - Tguess) > tol

    Tguess = T2guess;

    Cpm = A + (B*(Tguess+T1)/2) + (D/(Tguess*T1));

    T2guess = ((delH/R)/Cpm) + T1;

end

T2 = T2guess;
